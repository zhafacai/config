(defmacro map! (&rest args)
  "简易版 map!，支持 :leader 前缀 + 键-函数对，以及直接写键绑定。"
  (let* ((keys-and-funcs args)
         ;; 如果第一个是 :leader / :localleader 等前缀标记
         (prefix (when (keywordp (car keys-and-funcs))
                   (pop keys-and-funcs)))
         ;; 剩下的应该是 键 函数 键 函数 ... 的平铺列表
         bindings)

    ;; 把键值对收集起来
    (while keys-and-funcs
      (let ((key (pop keys-and-funcs))
            (fn  (pop keys-and-funcs)))
        (unless (and key fn)
          (error "map! 缺少对应的函数：看到 %S 后面没跟函数" key))
        (push `(define-key evil-normal-state-map
			   ,(if (keywordp prefix)
				(concat (symbol-name prefix) " " key)
			      key)
			   ,fn)
              bindings)))

    ;; 生成最终形式
    `(progn ,@(nreverse bindings))))

(defmacro map! (&rest args)
  "简易版 map!，支持 :leader / :localleader 前缀 + 键-函数对。"
  (let ((prefix nil)                    ; 前缀字符串（例如 "SPC" 或 ","）
        (keymaps '(evil-normal-state-map)) ; 默认绑定到 normal state
        bindings)

    ;; 第一步：解析前缀（如果有的话）
    (when (keywordp (car args))
      (pcase (pop args)
        (:leader
         (setq prefix (bound-and-true-p fc/leader-key)))
        (:localleader
         (setq prefix (bound-and-true-p fc/localleader-key)))
        (:prefix
         (setq prefix (pop args)))      ; 后面跟一个字符串，比如 "C-c"
        (_ (error "未知的 map! 前缀关键字：%S" (car args)))))

    ;; 第二步：收集键-函数对
    (while args
      (let ((key (pop args))
            (fn  (pop args)))
        (unless (and key fn)
          (error "map! 缺少对应的函数：看到键 %S 后面没有函数" key))

        ;; 如果 key 是字符串，就当作键描述符
        (let ((full-key
               (if prefix
                   (concat prefix key)   ; 加前缀
                 key)))                      ; 没有前缀就直接用

          (push `(define-key ,(car keymaps)  ; 目前只支持一个 map，可扩展
			     ,(if (stringp full-key)
				  `(kbd ,full-key)     ; 字符串 → kbd
				full-key)              ; 已经是 (kbd ...) 形式
			     ,fn)
                bindings))))

    ;; 生成最终形式
    `(progn ,@(nreverse bindings))))

(defun fc/emms-header-line ()
  "Show current track in header-line: metadata or filename-sans-ext + time."
  (when (and emms-player-playing-p
             (emms-playlist-current-selected-track))
    (let* ((track (emms-playlist-current-selected-track))
           (raw-desc (emms-track-description track))
           (display-desc
            (if (string-prefix-p "/" raw-desc)
                (file-name-base (emms-track-get track 'name))
              raw-desc))
           (elapsed emms-playing-time)
           (total (emms-track-get track 'info-playing-time)))
      (concat
       " 🎹 "
       display-desc
       (when (or elapsed total)
         (concat " ["
                 (if elapsed (fc/emms-format-time elapsed) "?")
                 (when total
                   (concat "/" (fc/emms-format-time total)))
                 "]"))))))

(defun fc/org-insert-link-dwim ()
  "Like `org-insert-link' but with personal dwim preferences."
  (interactive)
  (let* ((point-in-link (org-in-regexp org-link-any-re 1))
         (clipboard-url (when (string-match-p "^http" (current-kill 0))
                          (current-kill 0)))
         (region-content (when (region-active-p)
                           (buffer-substring-no-properties (region-beginning)
                                                           (region-end)))))
    (cond ((and region-content clipboard-url (not point-in-link))
           (delete-region (region-beginning) (region-end))
           (insert (org-make-link-string clipboard-url region-content)))
          ((and clipboard-url (not point-in-link))
           (insert (org-make-link-string
                    clipboard-url
                    (read-string "title: "
                                 (with-current-buffer (url-retrieve-synchronously clipboard-url)
                                   (dom-text (car
                                              (dom-by-tag (libxml-parse-html-region
                                                           (point-min)
                                                           (point-max))
                                                          'title))))))))
          (t
           (call-interactively 'org-insert-link)))))
(with-eval-after-load 'org
  (with-eval-after-load 'evil
    (evil-define-key nil org-mode-map
      (kbd "C-c C-l") 'fc/org-insert-link-dwim)))
